import pyautogui
import keyboard
import time

# Configuration: Adjust these values for your single-player farm setup
FARM_COORDINATES = (500, 300)  # Center of your farm (x, y) in screen coordinates
FARMING_INTERVAL = 2  # Seconds between actions
HOLD_DURATION = 0.5  # Seconds to hold mouse click

def groweasy():
    print("Starting GrowEasy... Press 'q' to stop.")
    print("For single-player Minecraft only. Do not use on multiplayer servers.")
    pyautogui.FAILSAFE = True  # Move mouse to top-left to stop script
    while not keyboard.is_pressed('q'):
        # Move to farm coordinates and perform action
        pyautogui.moveTo(*FARM_COORDINATES)
        pyautogui.mouseDown()
        time.sleep(HOLD_DURATION)
        pyautogui.mouseUp()
        time.sleep(FARMING_INTERVAL)

if __name__ == "__main__":
    print("GrowEasy: Educational tool for single-player Minecraft. Press 'q' to stop.")
    time.sleep(3)  # Delay to switch to Minecraft
    groweasy()
